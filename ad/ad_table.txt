create table company_info(
-- ads_c
c_idx int auto_increment primary key,
c_name varchar(50) not null,
c_dmn varchar(50),
c_num varchar(11) not null,
c_fax varchar(20),
c_pic varchar(20) not null,
c_email varchar(50) not null,
c_addr varchar(100) not null,
c_pr text,
c_logo text, not null
);
create table ads_info(
-- ads_info
ad_idx int auto_increment primary key,
ad_info_work varchar(20) not null,
ad_info_location varchar(20) not null,
ad_info_salary varchar(20) not null,
ad_info_school varchar(20) not null,
ad_info_worktype varchar(20) not null,
ad_info_ctype varchar(20) not null,
ad_info_career varchar(20),
ad_info_date varchar(20),
ad_info_essential varchar(50),
ad_info_preferential varchar(50),
ad_info_gender varchar(10),
ad_info_detail text not null,
start_date datetime not null,
end_date datetime not null
);
